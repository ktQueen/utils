/**
 * @module
 * @name is-empty
 * @description 判断`obj`是否为空
 * @author kt
 * @since 2019-01-03
 *
 *
 * @param {Object} obj 对象
 * @returns {Boolean} value 对象是否为空
 * @example
 *  var isEmpty=require('front-end-utils/is-empty');
 *  或
 *  import isEmpty from 'front-end-utils/is-empty';
 *  console.log(isEmpty({});
 *  ======》true
 */

module.exports = (obj) => {
    if (typeof obj == 'object') {
        if (Array.isArray(obj)) return obj.length <= 0;
        else return !Object.keys(obj).length;
    }
    return false;
};
