/**
 * @module
 * @name insert-string
 * @description 在字符串中插入新字符串
 * @author kt
 * @since 2019-01-03
 *
 *
 * @param {string} soure 源字符
 * @param {string} start 插入字符的位置
 * @param {string} newStr 需要插入的字符
 * @returns {string} 返回新生成的字符
 * @example
 *  var insertString=require('front-end-utils/insert-string');
 *  或
 *  import insertString from 'front-end-utils/insert-string';
 *  console.log(insertString('sadsd',2,'qqqq'));
 *  ======》saqqqqdsd
 */

module.exports = (soure, start, newStr)=>{
    return (soure.slice(0, start) + newStr + soure.slice(start));
};
