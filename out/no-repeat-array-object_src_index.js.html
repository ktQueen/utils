/**
 * @module
 * @name no-repeat-array-object
 * @description 数组对象去重
 * @author kt
 * @since 2018-12-18
 *
 * @param {array} arr -传入需要去重的数组对象
 * @return {array}  newArr -去重之后的数组对象
 * @example
 *  var noRepeatArrayObject=require('front-end-utils/no-repeat-array-object');
 *  或
 *  import noRepeatArrayObject from 'front-end-utils/no-repeat-array-object';
 *  console.log(noRepeatArrayObject([{'name':'1'},{'name':'1'},{'name':'2'}]));
 *  ======》[{'name':'1'},{'name':'2'}]
 */
let noRepeatArrayObject = {
    // 数组对象去重
    unRepeatObject(arr) {
        if (!(arr && arr.length > 0)) {
            return false
        }
        const arrL = arr.length;
        let tmepArr = [];
        // 将数组对象转成数组字符串
        let newStr = this.changeArrStr(arr);
        newStr.sort();
        // 数组去重
        for (let i = 0; i < newStr.length; i++) {
            if (newStr[i] !== tmepArr[tmepArr.length - 1]) {
                tmepArr.push(newStr[i]);
            }
        }
        let newArr = [];
        // 新数组字符串转成数组对象
        for (let i = 0; i < tmepArr.length; i++) {
            newArr.push(JSON.parse(tmepArr[i]));
        }
        return newArr;
    },
    // 数组内容对象转成字符串,去空,排序
    changeArrStr(arr) {
        var newArr = [];
        if (arr.length === 0) {
            return;
        } else {
            for (let i = 0; i < arr.length; i++) {
                let thisObj = this.sortObjectFun(arr[i]);
                let thisStr = JSON.stringify(thisObj);
                thisStr = thisStr.replace(/(\s|[\\t])/g, ''); // 去除空格及\t空白字符
                newArr.push(thisStr);
            }
        }
        return newArr;
    },
    // 对象排序
    sortObjectFun(obj) {
        let keyArr = [];// 对象的key
        for (var item in obj) {
            keyArr.push(item);
        }
        ;
        keyArr.sort(); // 降序
        let newObj = {};
        for (let i = 0; i < keyArr.length; i++) {
            newObj[keyArr[i]] = obj[keyArr[i]]
        }
        return newObj;
    }
};
module.exports = (arr) => {
    return noRepeatArrayObject.unRepeatObject(arr);
};
